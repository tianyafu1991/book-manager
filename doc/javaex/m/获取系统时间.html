获取系统时间

返回当前时间，格式yyyy-MM-dd HH:mm:ss
var time = javaex.getTime();
返回当前年份
var time = javaex.getTime("year");
返回当前月份
var time = javaex.getTime("month");
返回当前日
var time = javaex.getTime("day");
返回今日日期，格式yyyy-MM-dd
var time = javaex.getTime("today");
返回当日所属星期（1=周一，2=周二...7=周日）
var time = javaex.getTime("week");

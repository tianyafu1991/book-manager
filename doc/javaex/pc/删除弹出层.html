删除弹出层

删除专用弹出层，常用于列表删除数据
----------------------------------------------
代码
<button class="javaex-btn red" onclick="deleteDialog(this, 'xxxx')">点我测试</button>

<script>
	// obj为当前点击的元素对象
	function deleteDialog(obj, id) {
		// obj是必须的
		javaex.deleteDialog(obj, {
			content : "确定要删除么？",
			confirm : function() {
				console.log("获取变量值：" + id);
				// 如果你想阻止弹出层关闭，直接 return false; 即可
				// return false;
			}
		});
	}
</script>
参数说明
参数 默认值 说明 版本
content 弹出层内容
confirmName 确定 确定按钮名称
cancelName 取消 取消按钮名称
offsetTop 0 上偏移量
confirm 点击确定按钮后的回调函数
cancel 点击取消按钮后的回调函数
callback 打开弹窗后的回调函数
callback : function() {
	console.log("弹窗已打开");
}
